country_event = {
	id = into_the_unknown.1
	title = "west_african.24.t"
	desc = "west_african.24.d"
	picture = AFRICAN_EXPLORING_JUNGLE_eventPicture

	trigger = {
		OR = { 
			technology_group = sub_saharan
			technology_group = central_african
			technology_group = east_african
		}
		any_owned_province = {
			NOT = { has_province_modifier = used_to_tropical }
			any_empty_neighbor_province = {
				NOT = { has_province_modifier = used_to_tropical }
				OR = { 
					has_climate = tropical
					has_climate = arid
				}	
			}
		}
	}

	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 2
			NOT = { has_institution = feudalism }
		}

		modifier = {
			factor = 0.75
			has_institution = renaissance
		}

		modifier = {
			factor = 0.75
			has_institution = new_world_i
		}


	}

	# Send a party to explore
	option = {
		name = "west_african.24.a"
		add_manpower = -3
		random_owned_province = {
			limit = {
				any_empty_neighbor_province = {
					OR = {
						has_climate = tropical
						has_climate = arid
					}
				}
			}
			random_empty_neighbor_province = {
				limit = { 
					OR = {
						has_climate = tropical 
						has_climate = arid 
					}
				}				
				add_siberian_construction = 0
				add_province_modifier = {
					name = "used_to_tropical"
					duration = 120
				}
			}
		}
	}

	# Tell them to stay here
	option = {
		name = "west_african.24.b"
		add_prestige = -10
		add_adm_power = 20
	}
}
