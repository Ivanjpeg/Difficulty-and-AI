	title = "dif_manchu.EVTNAME1"
	desc = "dif_manchu.EVTDESC1"
	picture = ADVISOR_eventPicture

		ai = yes

			limit = {

			}

		}
		else_if = {
			limit = { 
				NOT = { exists = MCH }
				MJZ = { NOT = { num_of_cities = 6 } }
			}

				tooltip = root_is_most_powerful_manchuria_tribe
				OR = {
					primary_culture = manchu
					primary_culture = manchu_new
				}
				NOT = {
					any_country = {
						OR = {
							primary_culture = manchu
							primary_culture = manchu_new
						}
						OR = {
							army_size = root
							total_own_and_non_tributary_subject_development = root
						}
					}
				}
			}
		}
		else = {
			tag = MCH
		}
		OR = {
			is_year = 1500
			tag = MCH
		}
		NOT = { is_year = 1800 }
		NOT = { has_global_flag = nurhaci_spawned_flag }
	}
	
	mean_time_to_happen = {
		months = 500
		modifier = {
			factor = 2
			NOT = { is_year = 1525 }
		}
		modifier = {
			factor = 0.5
			tag = MCH
		}
		modifier = {
			factor = 0.5
			exists = CSH
		}
		modifier = {
			factor = 0.5
			is_year = 1550
		}
		modifier = {
			factor = 0.5
			MNG = { has_any_disaster = yes }
		}
		modifier = {
			factor = 0.2
			is_year = 1575
		}
	}

	immediate = {
		set_global_flag = nurhaci_spawned_flag
		if = {
			limit = {
				has_heir = yes
			}
			hidden_effect = {
				kill_heir = {}
			}
		}
	}

	option = {
		name = "flavor_mch.EVTOPTA8"
		define_ruler = {
			name = "Nurhachi"
			dynasty = "Aisin Gioro"
			adm = 6
			dip = 4
			mil = 6
			age = 20
		}
		hidden_effect = {
			set_ruler_flag = nurhaci_ruler_flag
			add_ruler_personality = strict_personality
			if = {
				limit = {
					ai = yes
				}
				add_ruler_modifier = {
					name = "great_unifier"
					duration = -1
					#desc = 
				}
			}
			set_ai_personality = {
				personality = ai_militarist
				locked = yes
			}
			log = "Nurhachi"
		}
	}
}

	id = dif_manchu.4
	title = "dif_manchu.EVTNAME4"
	desc = "dif_manchu.EVTDESC4"
	picture = LAND_MILITARY_eventPicture
	
	#fire_only_once = yes
	
	trigger = {
		OR = {
			AND = {
				is_at_war = no
				is_bankrupt = no
				NOT = { num_of_loans = 5 }
				total_development = 400
				NOT = { truce_with = MNG }
				NOT = { alliance_with = MNG }
				any_owned_province = { 
					any_neighbor_province = { 
						owned_by = MNG 
						OR = {
							is_claim = ROOT
							is_core = ROOT
						}	
					}
				}
			}
			war_with = MNG
		}
		OR = {
			NOT = { imperial_mandate = 5 }
			MNG = { has_any_disaster = yes }
			war_with = MNG
		}
		MNG = {
			is_emperor_of_china = yes
			ai = yes
		}
		exists = MNG
		NOT = { vassal_of = MNG }
		tag = MCH
		ai = yes
		#has_ruler_flag = nurhaci_ruler_flag
		NOT = { is_year = 1675 }
	}
	
	#immediate = {
	#	add_casus_belli = {
	#		target = MNG
	#		type = cb_border_war
	#		months = 24
	#	}
	#}

	mean_time_to_happen = {
		months = 100
		modifier = {
			factor = 0.01
			war_with = MNG
		}
		modifier = {
			factor = 0.25
			NOT = { imperial_mandate = 20 }
		}
		modifier = {
			factor = 0.25
			MNG = { has_any_disaster = yes }
		}
		modifier = {
			factor = 0.75
			exists = CSH
		}
	}
	
	option = {
		name = "dif_manchu.EVTOPTA4"
		if = {
			limit = {
				NOT = { has_country_modifier = manchu_conquest_of_china_ai }
				NOT = { has_country_flag = manchu_declered_war_dif }
			}
			add_country_modifier = {
				name = "manchu_conquest_of_china_ai"
				duration = 9125
				#desc = 
			}
		}
		if = {
			limit = {
				is_at_war = no
			}
			declare_war_with_cb = {
				who = MNG
				casus_belli = cb_take_mandate
			}
		}
		if = {
			limit = {
				num_of_loans = 1
				NOT = { has_country_flag = manchu_declered_war_dif }
			}
			add_treasury = 500
		}
		if = {
			limit = {
				MNG = { MCH = {	NOT = { mil_tech = PREV } } }
				NOT = { has_country_flag = manchu_declered_war_dif }
			}
			add_mil_tech = 1
		}
		set_country_flag = manchu_declered_war_dif
		log = "Macnu-Ming war"
	}
}
